    <script>
        // Airtable Config
        const AIRTABLE_BASE = "appP9QT33Umhfxm8y";
        const AIRTABLE_TOKEN = "pattnM5DzDIXR7IoO.8cdb605d2a9945131a9590d4f2877a5e1ce2fa29a04ad977b134859ca28321c2";
        const TABLE_EVENTS = "Events";

        const stations = [
            { name: "Nairobi Central Police", lat: -1.2833, lng: 36.8167 },
            { name: "Gigiri Police Station", lat: -1.2333, lng: 36.8000 },
            { name: "Kasarani Police Station", lat: -1.2167, lng: 36.9167 },
            { name: "Buruburu Police Station", lat: -1.3000, lng: 36.9167 }
        ];

        // Dark Mode Map Setup
        const map = L.map('map').setView([-1.286389, 36.817223], 12);
        const darkLayer = L.tileLayer('https://{s}.basemaps.cartocdn.com/dark_all/{z}/{x}/{y}{r}.png', { attribution: '© CARTO' }).addTo(map);
        const satLayer = L.tileLayer('https://server.arcgisonline.com/ArcGIS/rest/services/World_Imagery/MapServer/tile/{z}/{y}/{x}', { attribution: '© Esri' });
        let currentLayer = 'dark';
        
        function toggleMap() {
            if (currentLayer === 'dark') { map.removeLayer(darkLayer); satLayer.addTo(map); currentLayer = 'sat'; } 
            else { map.removeLayer(satLayer); darkLayer.addTo(map); currentLayer = 'dark'; }
        }

        let events = [];
        let eventMarkers = [];
        let heatLayer = null;

        async function fetchEvents() {
            try {
                const res = await fetch(`https://api.airtable.com/v0/${AIRTABLE_BASE}/${TABLE_EVENTS}`, {
                    headers: { Authorization: `Bearer ${AIRTABLE_TOKEN}` }
                });
                const data = await res.json();
                events = data.records.map(r => ({
                    id: r.id, name: r.fields.name, lat: r.fields.lat, lng: r.fields.lng,
                    type: r.fields.type, start: r.fields.start, estimated_end: r.fields.estimated_end,
                    closed_at: r.fields.closed_at
                }));
                updateDashboard();
                renderMapMarkers();
                renderList();
            } catch (e) { console.error("Telemetry Error", e); }
        }

        function updateDashboard() {
            const active = events.filter(e => !e.closed_at);
            document.getElementById('stat-active').innerText = active.length;
            document.getElementById('stat-demos').innerText = active.filter(e => e.type === 'Demonstration').length;
            document.getElementById('stat-closed').innerText = events.filter(e => e.closed_at).length;
        }

        function renderMapMarkers() {
            eventMarkers.forEach(m => map.removeLayer(m));
            eventMarkers = [];
            events.forEach(e => {
                const color = e.closed_at ? '#555' : (e.type === 'Demonstration' ? '#e74c3c' : '#f39c12');
                if (e.lat && e.lng) {
                    const marker = L.circleMarker([e.lat, e.lng], { radius: 8, color, fillColor: color, fillOpacity: 0.8 }).addTo(map);
                    marker.bindPopup(`<b>${e.name}</b><br>${e.type}<br>${e.closed_at ? 'Resolved' : 'Active'}`);
                    eventMarkers.push(marker);
                }
            });
        }

        function renderList() {
            const div = document.getElementById('events-list');
            div.innerHTML = '';
            events.forEach(e => {
                const isActive = !e.closed_at;
                div.innerHTML += `
                    <div class="event-card ${isActive ? '' : 'closed'}" style="border-left-color: ${isActive ? '#e74c3c' : '#555'}">
                        <h4>${e.name}</h4>
                        <div class="event-meta">Type: ${e.type} | Status: ${isActive ? '🟢 Active' : '⚪ Resolved'}</div>
                        ${isActive ? `<button onclick="closeEvent('${e.id}')" style="font-size:12px; padding:4px 8px;">Mark Resolved</button>` : ''}
                    </div>
                `;
            });
        }

        async function postEvent(data) {
            await fetch(`https://api.airtable.com/v0/${AIRTABLE_BASE}/${TABLE_EVENTS}`, {
                method: 'POST', headers: { Authorization: `Bearer ${AIRTABLE_TOKEN}`, 'Content-Type': 'application/json' },
                body: JSON.stringify({ fields: data })
            });
            fetchEvents();
        }

        async function reportEvent() {
            const sel = document.getElementById('locationSelect').value;
            if (!sel) return alert("Select Command Zone");
            const loc = stations[parseInt(sel)];
            const type = document.getElementById('eventType').value;
            const hours = parseFloat(document.getElementById('duration').value);
            
            await postEvent({
                name: loc.name, lat: loc.lat, lng: loc.lng, type,
                start: new Date().toISOString(), estimated_end: new Date(Date.now() + hours * 3600000).toISOString()
            });
        }

        async function closeEvent(id) {
            await fetch(`https://api.airtable.com/v0/${AIRTABLE_BASE}/${TABLE_EVENTS}/${id}`, {
                method: 'PATCH', headers: { Authorization: `Bearer ${AIRTABLE_TOKEN}`, 'Content-Type': 'application/json' },
                body: JSON.stringify({ fields: { closed_at: new Date().toISOString() } })
            });
            fetchEvents();
        }

        // --- EXPORT REPORT FEATURE ---
        function generateCSVReport() {
            if(events.length === 0) return alert("No operational data to export.");
            
            const headers = "ID,Location Name,Latitude,Longitude,Event Type,Start Time,Estimated End,Closed At\n";
            const rows = events.map(e => 
                `"${e.id}","${e.name}",${e.lat},${e.lng},"${e.type}","${e.start}","${e.estimated_end}","${e.closed_at || ''}"`
            ).join("\n");
            
            const blob = new Blob([headers + rows], { type: 'text/csv;charset=utf-8;' });
            const link = document.createElement("a");
            const url = URL.createObjectURL(blob);
            link.setAttribute("href", url);
            link.setAttribute("download", `KGM_Operations_Report_${new Date().toISOString().slice(0,10)}.csv`);
            link.style.visibility = 'hidden';
            document.body.appendChild(link);
            link.click();
            document.body.removeChild(link);
        }

        setInterval(fetchEvents, 5000);
        fetchEvents();
    </script>
